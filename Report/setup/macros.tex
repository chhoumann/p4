% Insert use cases easily with the \usecase command:
% \usecase{TITLE}{USE CASE}{OBJECTS}{FUNCTIONS}{uc:LABEL}{CAPTION}
\newcommand\tabularhead{
    \begin{tabular}{lp{11cm}}\hline
}
\newcommand\addheading[1]{
    \multicolumn{2}{c}{\textbf{\textit{#1}}}\\ \hline
}
\newcommand\addrow[2]{\textbf{#1}\begin{minipage}[t][][t]{11cm} \end{minipage}% 
   &\begin{minipage}[t][][t]{11cm}
    #2
    \end{minipage}\\
}
\newenvironment{usecaseenv}{
    \def\arraystretch{2}
    \tabularhead
}{  
    \hline\end{tabular}
    \def\arraystretch{1}
}

\let\oldFigureName\figurename %save the old definition of the caption
\newcommand{\usecase}[6]{
\vspace{0.5cm}
\renewcommand{\figurename}{Use case} %call caption figure name "Use case" instead
\begin{figure}[htbp]
    \begin{center}
        \begin{usecaseenv}
            \addheading{#1} 
            \addrow{Use case:}{#2}
            \addrow{Objects:}{#3}
            \addrow{Functions:}{#4}
        \end{usecaseenv}
    \end{center}
    \caption{#6}
    \label{#5}
\end{figure}
\renewcommand{\figurename}{\oldFigureName} %reset caption figure name
}
